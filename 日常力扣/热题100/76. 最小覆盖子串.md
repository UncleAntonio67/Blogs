```python
class Solution:
    def minWindow(self, s: str, t: str) -> str:
        cnt1 = Counter()
        cnt2 = Counter(t)
        ans_left, ans_right = -1,len(s)
        left = 0
        for right,c in enumerate(s):
            cnt1[c] += 1
            while cnt1 >= cnt2:
                if ans_right - ans_left > right - left:
                    ans_left, ans_right = left, right
                cnt1[s[left]] -= 1
                left += 1
        return "" if ans_left < 0 else s[ans_left:ans_right+1]
```
