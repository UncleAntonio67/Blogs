class Solution:
    def mergeKLists(self, lists: List[Optional[ListNode]]) -> Optional[ListNode]:
        def merge(l1,l2):
            dummy = p = ListNode(0)
            while l1 and l2:
                if l1.val < l2.val:
                    p.next = l1
                    l1 = l1.next
                else:
                    p.next = l2
                    l2 = l2.next
                p = p.next
            p.next = l1 or l2
            return dummy.next

        if not lists:
            return None
        res = lists[0]
        for i in range(1,len(lists)):
            res = merge(res, lists[i])
        return res
