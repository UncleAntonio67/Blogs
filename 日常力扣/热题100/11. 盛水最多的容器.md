class Solution:
    def maxArea(self, height: list[int]) -> int:
        slow = 0
        fast = len(height)-1
        area = 0
        while slow < fast:
            area = max(area, (fast-slow)*min(height[fast],height[slow]))
            if height[slow] > height[fast]:
                fast -= 1
            else:
                slow +=1
        return area
